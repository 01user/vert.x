def add?(obj)
  obj = SharedData.check_copy(obj)
  if !@j_set.contains(obj)
    @j_set.add(obj)
    self
  else
    nil
  end
end

def each(&block)
    iter = @j_set.iterator
    while iter.hasNext do
      block.call(iter.next)
    end
  end
