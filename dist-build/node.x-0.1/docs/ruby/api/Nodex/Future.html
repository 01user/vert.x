Overview

Represents an operation that may or may not have completed yet. It contains methods to determine it it has completed, failed or succeded, and allows a handler to be set which will be called when the operation completes, or fails.

def handler(proc = nil, &hndlr)
  hndlr = proc if proc
  @j_del.handler{ |j_def| hndlr.call(self) }
  self
end

def result
  if !@result_converted
    j_result = @j_del.result
    if (j_result == nil)
      @result = nil
    else
      if (@result_converter == nil)
        @result = j_result
      else
        @result = @result_converter.call(j_result)
      end
      @result_converted = true
    end
  end
  @result
end
