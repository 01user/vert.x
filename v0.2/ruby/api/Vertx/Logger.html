def error(msg, ex = nil)

  if ex == nil
    @j_del.error(msg)
  elsif ex.is_a? java.lang.Throwable
    @j_del.error(msg, ex)
  elsif ex.is_a? Exception
    @j_del.error(obj.message)
    trace = ''
    obj.backtrace.each { |line| trace << line << "\n" }
    @j_del.error(trace)
  end
end
